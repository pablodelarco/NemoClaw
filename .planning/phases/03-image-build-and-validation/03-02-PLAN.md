**Step 1: Upload the qcow2 image to OpenNebula**

From the build host (which should have OpenNebula CLI access), register the image:

```bash
ssh root@100.123.42.13 'bash -s' <<'UPLOAD'
IMAGE_PATH="/root/marketplace-community/apps-code/community-apps/output/nemoclaw.qcow2"

# Create the image in OpenNebula
oneimage create \
  --name "NemoClaw-v0.1.0-test" \
  --description "NemoClaw appliance test image" \
  --type OS \
  --driver qcow2 \
  --path "$IMAGE_PATH" \
  --datastore default

# Wait for image to be READY
echo "Waiting for image upload..."
for i in $(seq 1 60); do
    STATE=$(oneimage show "NemoClaw-v0.1.0-test" -x 2>/dev/null | grep -oP '(?<=<STATE>)\d+' || echo "0")
    if [ "$STATE" = "1" ]; then
        echo "Image is READY"
        break
    fi
    echo "  Upload in progress... (attempt $i/60)"
    sleep 10
done

oneimage show "NemoClaw-v0.1.0-test"
UPLOAD
```

**Step 2: Create a VM template with GPU passthrough**

The VM template must include PCI configuration for NVIDIA GPU passthrough (vendor 10de, class 0300 for VGA-compatible controllers). This addresses GPU-01.

```bash
ssh root@100.123.42.13 'bash -s' <<'TEMPLATE'
cat > /tmp/nemoclaw-template.txt <<'TMPL'
NAME = "NemoClaw-Test"
CPU = "2"
VCPU = "2"
MEMORY = "8192"

OS = [
  ARCH = "x86_64"
]

CPU_MODEL = [
  MODEL = "host-passthrough"
]

DISK = [
  IMAGE = "NemoClaw-v0.1.0-test"
]

NIC = [
  NETWORK = "vnet",
  MODEL = "virtio"
]

GRAPHICS = [
  LISTEN = "0.0.0.0",
  TYPE = "VNC"
]

CONTEXT = [
  NETWORK = "YES",
  SSH_PUBLIC_KEY = "$USER[SSH_PUBLIC_KEY]",
  ONEAPP_NEMOCLAW_API_KEY = "$ONEAPP_NEMOCLAW_API_KEY",
  ONEAPP_NEMOCLAW_MODEL = "nemotron-3-super-120b",
  ONEAPP_NEMOCLAW_SANDBOX_NAME = "nemoclaw"
]

PCI = [
  CLASS = "0300",
  DEVICE = "",
  VENDOR = "10de"
]
TMPL

onetemplate create /tmp/nemoclaw-template.txt
TEMPLATE
```

**IMPORTANT NOTES for the executor:**
- The `NIC NETWORK` name ("vnet") may need to be adjusted to match the actual virtual network name in the OpenNebula deployment. Run `onevnet list` to find available networks.
- The `ONEAPP_NEMOCLAW_API_KEY` must be set to a real NVIDIA API key when instantiating. The user will need to provide this.
- The PCI section uses CLASS=0300 (VGA) and VENDOR=10de (NVIDIA) to auto-select any available NVIDIA GPU on the host. If no GPU is available, remove the PCI section and test without GPU (remote inference only).

**Step 3: Instantiate the VM**

```bash
ssh root@100.123.42.13 'bash -s' <<'INSTANTIATE'
# The user must provide their NVIDIA API key here
# Replace YOUR_NVIDIA_API_KEY with the actual key
onetemplate instantiate "NemoClaw-Test" \
  --name "nemoclaw-validation" \
  --context "ONEAPP_NEMOCLAW_API_KEY=YOUR_NVIDIA_API_KEY"

# Wait for VM to reach RUNNING state
echo "Waiting for VM to boot..."
for i in $(seq 1 30); do
    STATE=$(onevm show "nemoclaw-validation" -x 2>/dev/null | grep -oP '(?<=<STATE>)\d+' || echo "0")
    LCM_STATE=$(onevm show "nemoclaw-validation" -x 2>/dev/null | grep -oP '(?<=<LCM_STATE>)\d+' || echo "0")
    if [ "$STATE" = "3" ] && [ "$LCM_STATE" = "3" ]; then
        echo "VM is RUNNING"
        break
    fi
    echo "  VM booting... state=$STATE lcm=$LCM_STATE (attempt $i/30)"
    sleep 10
done

# Get VM IP address
VM_IP=$(onevm show "nemoclaw-validation" -x 2>/dev/null | grep -oP '(?<=<IP>)[^<]+' | head -1)
echo "VM IP: $VM_IP"
INSTANTIATE
```

**Step 1: Wait for VM contextualization to complete (up to 5 minutes)**

The VM runs service_configure then service_bootstrap on first boot. This may take 2-5 minutes.

```bash
# Get VM IP from previous task output, or query it
VM_IP=$(ssh root@100.123.42.13 'onevm show "nemoclaw-validation" -x 2>/dev/null | grep -oP "(?<=<IP>)[^<]+" | head -1')
echo "VM IP: $VM_IP"

# Wait for SSH to become available (ACC-01)
for i in $(seq 1 30); do
    if ssh -o StrictHostKeyChecking=no -o ConnectTimeout=5 root@"$VM_IP" 'echo SSH_OK' 2>/dev/null; then
        echo "SSH access confirmed"
        break
    fi
    echo "  Waiting for SSH... (attempt $i/30)"
    sleep 10
done
```

**Step 2: Validate SSH key-based authentication (ACC-01)**

```bash
ssh -o StrictHostKeyChecking=no root@"$VM_IP" 'bash -s' <<'SSH_CHECK'
echo "=== SSH Access Validation (ACC-01) ==="
echo "User: $(whoami)"
echo "Hostname: $(hostname)"
echo "SSH config:"
grep -E "PasswordAuthentication|PermitRootLogin" /etc/ssh/sshd_config | grep -v "^#"
echo "=== SSH OK ==="
SSH_CHECK
```

**Step 3: Validate GPU passthrough (GPU-01, GPU-03)**

```bash
ssh -o StrictHostKeyChecking=no root@"$VM_IP" 'bash -s' <<'GPU_CHECK'
echo "=== GPU Validation (GPU-01, GPU-03) ==="

echo "--- lspci NVIDIA devices ---"
lspci | grep -i nvidia || echo "No NVIDIA devices in lspci"

echo "--- /dev/nvidia* devices ---"
ls -la /dev/nvidia* 2>/dev/null || echo "No /dev/nvidia* device nodes"

echo "--- nvidia-smi ---"
nvidia-smi 2>/dev/null || echo "nvidia-smi failed or no GPU"

echo "--- Docker GPU test ---"
docker run --rm --gpus all nvidia/cuda:12.0-base nvidia-smi 2>/dev/null || echo "Docker GPU test failed (may be expected if no GPU passthrough)"

echo "--- GPU detection flag from appliance ---"
cat /opt/one-appliance/nemoclaw.conf 2>/dev/null | grep GPU || echo "No GPU config found"

echo "=== GPU Check Complete ==="
GPU_CHECK
```

**Step 4: Validate NemoClaw CLI interaction (ACC-03)**

```bash
ssh -o StrictHostKeyChecking=no root@"$VM_IP" 'bash -s' <<'CLI_CHECK'
echo "=== NemoClaw CLI Validation (ACC-03) ==="

echo "--- NemoClaw version ---"
nemoclaw --version 2>/dev/null || echo "nemoclaw CLI not found"

echo "--- NemoClaw status ---"
nemoclaw status 2>/dev/null || echo "nemoclaw status failed"

echo "--- NemoClaw sandbox list ---"
nemoclaw list 2>/dev/null || echo "nemoclaw list failed"

echo "--- NemoClaw sandbox status ---"
nemoclaw nemoclaw status 2>/dev/null || echo "sandbox status check failed"

echo "--- Docker containers ---"
docker ps 2>/dev/null || echo "docker ps failed"

echo "--- Node.js version ---"
node --version 2>/dev/null || echo "node not found"

echo "=== CLI Check Complete ==="
CLI_CHECK
```
